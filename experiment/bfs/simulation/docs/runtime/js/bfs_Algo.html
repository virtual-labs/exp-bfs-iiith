2.1 Class of Graph

class Graph {
    constructor() {
        this.edges = {};
        this.nodes = [];
        this.visited = new Array(100);
        this.visited.fill(0);
    }

    addNode(node) {
        this.nodes.push(node);
        this.edges[node] = [];
    }

    addEdge(node1, node2, id) {
        this.edges[node1].push({ node: node2, id: id });
    }
